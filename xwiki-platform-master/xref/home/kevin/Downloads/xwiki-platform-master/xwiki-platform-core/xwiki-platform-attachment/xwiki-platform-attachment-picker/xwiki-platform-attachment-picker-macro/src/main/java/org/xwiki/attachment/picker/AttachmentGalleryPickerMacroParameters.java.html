20  package org.xwiki.attachment.picker;
21  
22  import java.util.ArrayList;
23  import java.util.List;
24  
25  import org.xwiki.attachment.picker.internal.AttachmentGalleryPickerMacro;
26  import org.xwiki.model.reference.DocumentReference;
27  import org.xwiki.properties.annotation.PropertyDescription;
28  import org.xwiki.properties.annotation.PropertyDisplayType;
29  import org.xwiki.properties.annotation.PropertyName;
30  import org.xwiki.stability.Unstable;
31  
32  /**
33   * Parameters for the {@link AttachmentGalleryPickerMacro} Macro.
34   *
35   * @version $Id: b83cdcc40a28410ade3e8470bddff76a7d7ed099 $
36   * @since 14.4RC1
37   */
38  @Unstable
39  public class AttachmentGalleryPickerMacroParameters
40  {
41      private String id;
42  
43      private List<String> filter = new ArrayList<>();
44  
45      private Integer limit = 20;
46  
47      private String targetDocumentReference;
48  
49      /**
50       * @return the id of the attachment picker macro
51       */
52      public String getId()
53      {
54          return this.id;
55      }
56  
57      /**
58       * @param id the id of the attachment picker macro
59       */
60      @PropertyName("Id")
61      @PropertyDescription("The id of the attachment picker macro")
62      public void setId(String id)
63      {
64          this.id = id;
65      }
66  
67      /**
68       * @return the supported types of attachments
69       */
70      public List<String> getFilter()
71      {
72          return this.filter;
73      }
74  
75      /**
76       * @param filter the supported types of attachments
77       */
78      @PropertyName("Filter")
79      @PropertyDescription("A list of mimetypes to filter the attachments by (e.g. image/*,image/jpeg)")
80      public void setFilter(List<String> filter)
81      {
82          this.filter = filter;
83      }
84  
85      /**
86       * @return the maximum number of attachments to display
87       */
88      public Integer getLimit()
89      {
90          return this.limit;
91      }
92  
93      /**
94       * @param limit the maximum number of attachments to display
95       */
96      @PropertyName("Limit")
97      @PropertyDescription("The maximum number of attachments to display")
98      public void setLimit(Integer limit)
99      {
100         this.limit = limit;
101     }
102 
103     /**
104      * @return the reference to the document that will be set as the current document to display the macro content
105      * @since 14.10.2
106      * @since 15.0RC1
107      */
108     @Unstable
109     public String getTarget()
110     {
111         return this.targetDocumentReference;
112     }
113 
114     /**
115      * @param targetDocumentReference refer to {@link #getTarget()}
116      * @since 14.10.2
117      * @since 15.0RC1
118      */
119     @PropertyDescription("The reference to the document serving as the current document")
120     @PropertyDisplayType(DocumentReference.class)
121     @Unstable
122     public void setTarget(String targetDocumentReference)
123     {
124         this.targetDocumentReference = targetDocumentReference;
125     }
126 }
